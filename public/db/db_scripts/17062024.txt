CREATE TABLE employee_contacts ( id INT UNSIGNED AUTO_INCREMENT PRIMARY KEY, employee_id INT UNSIGNED NOT NULL, employee_contact_name VARCHAR(255) NOT NULL, employee_address TEXT NOT NULL, employee_phone VARCHAR(255) NOT NULL, employee_mobile VARCHAR(255) NOT NULL, employee_email VARCHAR(255) NOT NULL, employee_relationship VARCHAR(255) NOT NULL, created_at TIMESTAMP NULL DEFAULT NULL, updated_at TIMESTAMP NULL DEFAULT NULL, FOREIGN KEY (employee_id) REFERENCES users(id) ON DELETE CASCADE ) ENGINE=InnoDB;
